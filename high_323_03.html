# 딕셔너리에 1차원 데이터 표현하기
T_size = {'내빈':95, '미송':85, '민수':100, '준서':95, '지환':90 }
print('우리 반 모든 학생의 상의 치수:', T_size)

# 키를 이용해 딕셔너리의 값 가져오기
print('내빈이의 상의 치수:', T_size['내빈'])

# 키를 이용해 딕셔너리의 값 수정하기
T_size['미송'] = 90
print('미송의 수정된 사이즈:', T_size['미송'])

# 딕셔너리에 새로운 키:값 쌍을 저장하기
T_size['재우'] = 105
print('우리 반 모든 학생의 상의 치수:', T_size)
